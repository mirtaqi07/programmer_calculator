    # This code is in C#.
    # It converts form:
        # Decimal To:
                # Binary
                # Octal
                # Hexa
        # Binary To:
                # Decimal
                # Octal
                # Hexa
        # Octal To:
                # Decimal
                # Binary
                # Hexa
        # Hexa To:
                # Decimal
                # Binary
                # Octal
